#include<stdio.h>
#include<stdlib.h>

int main(){
    int i,n,s, flag;
    flag=0;
    printf("Enter size of array");
    scanf("%d",&n);
    int a[n];
    for(i=0;i<n;i++){
        scanf("%d",&a[i]);

    }
    printf("element for the search");
    scanf("%d",&s);

    for(i=0;i<n;i++){
        if(a[i]==s){
            printf("element in index:%d",i);
            flag=1;
        }
    }
    if(flag==0){
        printf("not found");
    }
}

#include<stdio.h>
#include<stdlib.h>

int main(){
    int i,n,c;
    printf("Enter size of array");
    scanf("%d",&n);
    int a[n];
    for(i=0;i<n;i++){
        scanf("%d",&a[i]);

    }

    for(i=0;i<n;i++){
        for(int j=i+1;j<n;j++){
            if(a[i]== a[j]){
                c=c+1;
                printf("no:%d is repeated %d times",a[i],c);
            }
        }
        c=0;
    }


}
